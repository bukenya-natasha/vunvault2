let scanType='full',scanActive=false,findings=[],findingCount=0;

function hashStr(s){let h=0x811c9dc5;for(let i=0;i<s.length;i++){h^=s.charCodeAt(i);h=Math.imul(h,0x01000193);}return h>>>0;}

function getFindings(domain){
  const clean=domain.toLowerCase().replace(/^https?:\/\//,'').replace(/\/.*/,'');
  const checks=scanType==='quick'?CHECKS.filter(c=>c.cat==='Security Headers'||c.cat==='TLS / SSL'):
    scanType==='owasp'?CHECKS.filter(c=>c.cat!=='Email / DNS'&&c.cat!=='TLS / SSL'):CHECKS;
  const results=[];
  checks.forEach((c,i)=>{
    const seed=hashStr(clean+c.id+i);
    if((seed%1000)/1000<c.w) results.push(c);
  });
  if(results.length<2){
    const extras=checks.filter(c=>!results.find(r=>r.id===c.id));
    extras.sort((a,b)=>b.w-a.w);
    while(results.length<2&&extras.length) results.push(extras.shift());
  }
  if(results.length>10) results.length=10;
  return results;
}

function calcScore(f){
  const d={CRITICAL:22,HIGH:14,MEDIUM:7,LOW:3};
  return Math.max(0,Math.min(100,100-f.reduce((s,c)=>s+(d[c.sev]||0),0)));
}

function drawGauge(score){
  const r=56,cx=70,cy=70;
  const toRad=d=>(d*Math.PI)/180;
  const pt=d=>({x:cx+r*Math.cos(toRad(d)),y:cy+r*Math.sin(toRad(d))});
  const sA=215,sweep=290;
  const s=pt(sA),e=pt(sA+sweep);
  const lA=sweep>180?1:0;
  document.getElementById('gaugeTrack').setAttribute('d',`M${s.x},${s.y}A${r},${r},0,${lA},1,${e.x},${e.y}`);
  if(score>0){
    const endD=sA+sweep*(score/100);
    const pe=pt(endD);
    const laF=sweep*(score/100)>180?1:0;
    const col=score>=80?'#5fd988':score>=60?'#4dd9e8':score>=40?'#ff8a3d':'#ff4d4d';
    document.getElementById('gaugeFill').setAttribute('d',`M${s.x},${s.y}A${r},${r},0,${laF},1,${pe.x},${pe.y}`);
    document.getElementById('gaugeFill').setAttribute('stroke',col);
    document.getElementById('gaugeScore').textContent=score;
    const lbl=score>=80?'SECURE':score>=60?'MODERATE':score>=40?'AT RISK':'CRITICAL';
    document.getElementById('gaugeLabel').textContent=lbl;
    document.getElementById('gaugeLabel').setAttribute('fill',col);
  }
}

function addTermLine(text,cls){
  const body=document.getElementById('terminalBody');
  const line=document.createElement('div');
  line.className='t-line '+(cls||'t-step');
  line.textContent=text;
  body.appendChild(line);
  body.scrollTop=body.scrollHeight;

}

function addFinding(check){
  findingCount++;
  document.getElementById('findingCount').textContent=findingCount+' found';

}

async function sleep(ms){return new Promise(r=>setTimeout(r,ms));}

async function startScan(){
  if(scanActive) return;
  const raw=document.getElementById('domainInput').value.trim();
  const domain=raw||'demo-target.co.ke';
  document.getElementById('domainInput').value=domain;

  scanActive=true;
  findings=[];
  findingCount=0;

  document.getElementById('scanBtn').disabled=true;
  document.getElementById('scanBtn').textContent='Scanning…';
  document.getElementById('terminalBody').innerHTML='';
  document.getElementById('resultsBody').innerHTML=`<div class="results-idle"><p>Scanning ${domain}…</p></div>`;
  document.getElementById('findingCount').style.display='none';
  document.getElementById('scorePanel').className='score-panel';
  document.getElementById('ctaPanel').className='cta-panel';
  document.getElementById('scanArea').className='scan-area visible';
  document.getElementById('progressFill').style.width='0%';
  document.getElementById('progressPct').textContent='0%';
  document.getElementById('terminalTitle').textContent=`vunvault-scanner — ${domain}`;
  document.getElementById('gaugeScore').textContent='--';
  document.getElementById('gaugeLabel').textContent='SCORING';
  document.getElementById('gaugeFill').setAttribute('d','');

  const allFindings=getFindings(domain);
  const checks=scanType==='quick'?CHECKS.filter(c=>c.cat==='Security Headers'||c.cat==='TLS / SSL'):
    scanType==='owasp'?CHECKS.filter(c=>c.cat!=='Email / DNS'&&c.cat!=='TLS / SSL'):CHECKS;

  addTermLine(`[VUNVAULT] Starting ${scanType} scan of ${domain}`,'t-cat');
  addTermLine(`[VUNVAULT] ${checks.length} checks queued`,'t-step');
  await sleep(400);

  let lastCat='';
  for(let i=0;i<checks.length;i++){
    const c=checks[i];
    if(c.cat!==lastCat){
      addTermLine(`\n▸ ${c.cat.toUpperCase()}`,'t-cat');
      lastCat=c.cat;
    }
    for(const step of c.steps){
      addTermLine(`  ${step}`,'t-step');
      await sleep(55+Math.random()*45);
    }
    const isFinding=allFindings.find(f=>f.id===c.id);
    if(isFinding){
      const cls=isFinding.sev==='CRITICAL'?'t-critical':'t-find';
      addTermLine(`  ⚠ ${isFinding.sev} — ${isFinding.title}`,cls);
      findings.push(isFinding);
      addFinding(isFinding);
    } else {
      addTermLine('  ✓ Pass','t-pass');
    }
    const pct=Math.round(((i+1)/checks.length)*100);
    document.getElementById('progressFill').style.width=pct+'%';
    document.getElementById('progressPct').textContent=pct+'%';
    await sleep(90+Math.random()*70);
  }

  await sleep(300);
  const score=calcScore(findings);
  const crit=findings.filter(f=>f.sev==='CRITICAL').length;
  const high=findings.filter(f=>f.sev==='HIGH').length;
  const med=findings.filter(f=>f.sev==='MEDIUM').length;

  addTermLine(`\n[VUNVAULT] Scan complete — ${findings.length} findings, score: ${score}/100`,'t-done');
  document.getElementById('scorePanel').className='score-panel show';
  document.getElementById('countCrit').textContent=crit;
  document.getElementById('countHigh').textContent=high;
  document.getElementById('countMed').textContent=med;
  drawGauge(score);

  await sleep(400);
  document.getElementById('ctaPanel').className='cta-panel show';
  document.getElementById('scanBtn').disabled=false;
  document.getElementById('scanBtn').textContent='Run Demo Scan';
  scanActive=false;
}

  document.getElementById('scanArea').className='scan-area';

  document.getElementById('domainInput').value='';
  document.getElementById('domainInput').focus();
  findings=[];findingCount=0;scanActive=false;
}

document.getElementById('domainInput').addEventListener('keydown',e=>{if(e.key==='Enter'){e.preventDefault();startScan();}});
